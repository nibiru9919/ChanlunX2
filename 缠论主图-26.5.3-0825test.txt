{绑定为2号函数}
FRAC:=TDXDLL2(2,H,L,0);{标准笔}
NOTEXT画上升笔2:DRAWLINE(FRAC=-1,L,FRAC=+1,H,0), DOTLINE,COLORYELLOW;
NOTEXT画下降笔2:DRAWLINE(FRAC=+1,H,FRAC=-1,L,0), DOTLINE, COLORYELLOW;

BIZG:=TDXDLL2(5,FRAC,H,L);{输出BI中枢高}
BIZD:=TDXDLL2(6,FRAC,H,L);{输出BI中枢低}
BISE:=TDXDLL2(7,FRAC,H,L);{输出BI中枢开始和结束}

NOTEXT_BIZG:IF(BIZG,BIZG,DRAWNULL),COLORYELLOW;{画BI中枢高}
NOTEXT_BIZD:IF(BIZD,BIZD,DRAWNULL),COLORYELLOW;{画BI中枢低}
NOTEXT_BISE:STICKLINE(BISE,BIZD,BIZG,0,0),COLORYELLOW;{画BI中枢起始结束};

DUAN1:=TDXDLL2(3,FRAC,H,L);{计算段的端点,3改成4是1+1终结画法}
NOTEXT画上升段1:DRAWLINE(DUAN1=-1,L,DUAN1=+1,H,0), COLORFF8000;
NOTEXT画下降段1:DRAWLINE(DUAN1=+1,H,DUAN1=-1,L,0), COLORFF8000;

DUANZG1:=TDXDLL2(5,DUAN1,H,L);{输出段中枢高}
DUANZD1:=TDXDLL2(6,DUAN1,H,L);{输出段中枢低}
DUANSE1:=TDXDLL2(7,DUAN1,H,L);{输出段中枢开始和结束}

NOTEXT_DDUANZG1:IF(DUANZG1,DUANZG1,DRAWNULL),COLORFF8000;{画段中枢高}
NOTEXT_DDUANZD1:IF(DUANZD1,DUANZD1,DRAWNULL),COLORFF8000;{画段中枢低}
NOTEXT_DDUANSE1:STICKLINE(DUANSE1,DUANZD1,DUANZG1,0,0),COLORFF8000;{画段中枢起始结束};

{新增缠论买卖点识别}

{一买：下跌趋势中，次级别走势跌破最后一个中枢后出现背驰的点}
{基于笔的背驰识别 - 简化版实现}
ZSBJ:=IF(BISE,BIZG+BIZD)/2,DRAWNULL;{中枢中轴线}
ZSBD:=IF(BISE,BIZG-BIZD),DRAWNULL;{中枢波动幅度}

{一买判断：笔向下突破中枢后形成背驰}
YB1:=(FRAC=-1 AND REF(FRAC,1)=1 AND L<REF(BIZD,1) AND 
     (C>REF(C,1) OR (REF(C,2)>REF(C,3) AND C<REF(C,1))));{简化背驰判断}
DRAWTEXT(YB1,L*0.99,'一买'),COLORRED;

{二买：一买后回调不破一买点，形成第二类买点}
YB2:=(FRAC=-1 AND REF(FRAC,1)=1 AND L>REF(L,5) AND REF(YB1,5));{回调不破前5根笔低点，且前5根笔有出现一买}
DRAWTEXT(YB2,L*0.99,'二买'),COLORMAGENTA;

{三买：向上突破中枢后回调不破中枢上沿}
YB3:=(FRAC=1 AND REF(FRAC,1)=-1 AND H>REF(BIZG,1) AND 
     REF(FRAC,2)=1 AND REF(L,1)>REF(BIZG,1));{突破中枢后回调不破中枢上沿}
DRAWTEXT(YB3,L*0.99,'三买'),COLORBLUE;

{类二买：中枢下沿附近的买点}
LEB2:=(FRAC=-1 AND ABS(L-REF(BIZD,1))/REF(BIZD,1)<0.005 AND 
       REF(FRAC,1)=1 AND REF(H,1)<REF(BIZD,1));{在中枢下沿附近形成笔}
DRAWTEXT(LEB2,L*0.99,'类二买'),COLORYELLOW;

{一卖：上涨趋势中，次级别走势突破最后一个中枢后出现顶背驰的点}
YS1:=(FRAC=1 AND REF(FRAC,1)=-1 AND H>REF(BIZG,1) AND 
      (C<REF(C,1) OR (REF(C,2)<REF(C,3) AND C>REF(C,1))));{简化顶背驰判断}
DRAWTEXT(YS1,H*1.01,'一卖'),COLORGREEN;

{二卖：一卖后反弹不破一卖点，形成第二类卖点}
YS2:=(FRAC=1 AND REF(FRAC,1)=-1 AND H<REF(H,5) AND REF(YS1,5));{反弹不破前5根笔高点，且前5根笔有出现一卖}
DRAWTEXT(YS2,H*1.01,'二卖'),COLORCYAN;

{三卖：向下突破中枢后反弹不破中枢下沿}
YS3:=(FRAC=-1 AND REF(FRAC,1)=1 AND L<REF(BIZD,1) AND 
      REF(FRAC,2)=-1 AND REF(H,1)<REF(BIZD,1));{跌破中枢后反弹不破中枢下沿}
DRAWTEXT(YS3,H*1.01,'三卖'),COLORWHITE;

{类二卖：中枢上沿附近的卖点}
LES2:=(FRAC=1 AND ABS(H-REF(BIZG,1))/REF(BIZG,1)<0.005 AND 
       REF(FRAC,1)=-1 AND REF(L,1)>REF(BIZG,1));{在中枢上沿附近形成笔}
DRAWTEXT(LES2,H*1.01,'类二卖'),COLORLIRED;

{多级别适配提醒：此指标可在日线、30分钟、5分钟、1分钟级别使用}
{注意：不同级别背驰判断参数可能需要调整以适应不同波动特性}
